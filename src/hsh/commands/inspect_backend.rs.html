4//! `hsh inspect-backend` — show the effective crypto route for a preset.
5//!
6//! This is the operator-facing self-check: given the preset a service is
7//! about to deploy under, what does the binary actually do? Answers:
8//!
9//! - Which [`Backend`] does the preset declare? (`Native` / `Fips140Required`)
10//! - Can this build satisfy the declared backend? (i.e. is the `fips`
11//!   feature compiled in *and* a validated backend wired through?)
12//! - Which primary algorithm will new hashes be minted under?
13//! - Is the `pepper` feature compiled in for this binary?
14//! - Build provenance: hsh crate version, rustc version, target triple,
15//!   profile (debug/release).
16//!
17//! [`Backend`]: hsh::Backend
18
19use anyhow::{Context, Result};
20
21use crate::cli::{InspectBackendArgs, PresetPolicy};
22use crate::io::print_kv;
23use hsh::policy::Policy;
24
25const HSH_VERSION: &str = env!("CARGO_PKG_VERSION");
26const TARGET_TRIPLE: &str = env!("HSH_TARGET_TRIPLE");
27const PROFILE: &str = env!("HSH_PROFILE");
28const RUSTC_VERSION: &str = env!("HSH_RUSTC_VERSION");
29
30pub(crate) fn run(args: InspectBackendArgs, json: bool) -> Result<()> {
31    let preset_name = match args.policy {
32        PresetPolicy::Owasp => "owasp_minimum_2025",
33        PresetPolicy::Rfc9106 => "rfc9106_first_recommended",
34        PresetPolicy::Fips => "fips_140_pbkdf2",
35    };
36    let policy = match args.policy {
37        PresetPolicy::Owasp => Policy::owasp_minimum_2025(),
38        PresetPolicy::Rfc9106 => Policy::rfc9106_first_recommended(),
39        PresetPolicy::Fips => Policy::fips_140_pbkdf2(),
40    };
41
42    let backend_label = if policy.backend().is_fips() {
43        "Fips140Required"
44    } else {
45        "Native"
46    };
47    let primary = format!("{:?}", policy.primary());
48    let fips_available = hsh::Backend::fips_available_in_build();
49    let pepper_feature = cfg!(feature = "pepper");
50
51    let satisfies = if policy.backend().is_fips() {
52        // FIPS policy is satisfied iff a FIPS-capable build is present.
53        fips_available
54    } else {
55        // Native is always satisfiable.
56        true
57    };
58    let satisfies_label = if satisfies {
59        "satisfied"
60    } else {
61        "unsatisfied (build cannot provide a FIPS-validated route)"
62    };
63
64    let pairs: Vec<(String, serde_json::Value)> = vec![
65        ("preset".into(), preset_name.into()),
66        ("backend".into(), backend_label.into()),
67        ("primary_algorithm".into(), primary.into()),
68        (
69            "fips_available_in_build".into(),
70            serde_json::Value::Bool(fips_available),
71        ),
72        (
73            "pepper_feature_compiled".into(),
74            serde_json::Value::Bool(pepper_feature),
75        ),
76        ("readiness".into(), satisfies_label.into()),
77        ("hsh_cli_version".into(), HSH_VERSION.into()),
78        ("rustc".into(), RUSTC_VERSION.into()),
79        ("target_triple".into(), TARGET_TRIPLE.into()),
80        ("profile".into(), PROFILE.into()),
81    ];
82
83    emit(json, &pairs).context("emit inspect-backend output")
84}
85
86fn emit(
87    json: bool,
88    pairs: &[(String, serde_json::Value)],
89) -> Result<()> {
90    let kv: Vec<(&str, &serde_json::Value)> =
91        pairs.iter().map(|(k, v)| (k.as_str(), v)).collect();
92    print_kv(json, &kv)
93}
