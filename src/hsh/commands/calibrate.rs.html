4//! `hsh calibrate` — find parameters that hit a target wall-time.
5//!
6//! Walks a small parameter ladder for the chosen algorithm, measures
7//! `hsh::api::hash` wall-time, and reports the params closest to the
8//! caller's target. Useful when sizing servers for new deployments.
9
10use anyhow::Result;
11use std::time::Instant;
12
13use crate::cli::{AlgoArg, CalibrateArgs};
14use crate::io::print_kv;
15use hsh::algorithms::bcrypt::BcryptParams;
16use hsh::algorithms::pbkdf2::{Pbkdf2Params, Prf};
17use hsh::algorithms::scrypt::ScryptParams;
18use hsh::policy::{Policy, PolicyBuilder, PrimaryAlgorithm};
19
20const PROBE_PASSWORD: &str = "calibration-probe-1234567890";
21
22pub(crate) fn run(args: CalibrateArgs, json: bool) -> Result<()> {
23    let target = u128::from(args.target_ms);
24    let mut best: Option<(String, u128)> = None;
25    let mut ladder: Vec<LadderEntry> = Vec::new();
26
27    match args.algorithm {
28        AlgoArg::Argon2id | AlgoArg::Argon2i | AlgoArg::Argon2d => {
29            // Ladder over memory cost (KiB), holding t=2, p=1.
30            for m in [4_096u32, 8_192, 19_456, 32_768, 65_536, 131_072]
31            {
32                let policy = PolicyBuilder::from_preset(
33                    &Policy::owasp_minimum_2025(),
34                )
35                .primary(PrimaryAlgorithm::Argon2id)
36                .argon2(argon2::Params::new(m, 2, 1, Some(32)).unwrap())
37                .build()
38                .unwrap();
39                let took = time_hash(&policy);
40                let params = format!("argon2id m={m} t=2 p=1");
41                ladder.push(LadderEntry::new(&params, took, target));
42                consider(&mut best, params, took, target);
43            }
44        }
45        AlgoArg::Bcrypt => {
46            for cost in 4u32..=14 {
47                let policy = PolicyBuilder::from_preset(
48                    &Policy::owasp_minimum_2025(),
49                )
50                .primary(PrimaryAlgorithm::Bcrypt)
51                .bcrypt(BcryptParams::new(cost))
52                .build()
53                .unwrap();
54                let took = time_hash(&policy);
55                let params = format!("bcrypt cost={cost}");
56                ladder.push(LadderEntry::new(&params, took, target));
57                consider(&mut best, params, took, target);
58            }
59        }
60        AlgoArg::Scrypt => {
61            for log_n in 8u8..=17 {
62                let policy = PolicyBuilder::from_preset(
63                    &Policy::owasp_minimum_2025(),
64                )
65                .primary(PrimaryAlgorithm::Scrypt)
66                .scrypt(ScryptParams {
67                    log_n,
68                    r: 8,
69                    p: 1,
70                    dk_len: 32,
71                })
72                .build()
73                .unwrap();
74                let took = time_hash(&policy);
75                let params = format!("scrypt log_n={log_n} r=8 p=1");
76                ladder.push(LadderEntry::new(&params, took, target));
77                consider(&mut best, params, took, target);
78            }
79        }
80        AlgoArg::Pbkdf2 => {
81            for iters in [
82                10_000u32, 50_000, 100_000, 200_000, 400_000, 600_000,
83                1_000_000,
84            ] {
85                let policy = PolicyBuilder::from_preset(
86                    &Policy::owasp_minimum_2025(),
87                )
88                .primary(PrimaryAlgorithm::Pbkdf2)
89                .pbkdf2(Pbkdf2Params {
90                    prf: Prf::Sha256,
91                    iterations: iters,
92                    dk_len: 32,
93                })
94                .build()
95                .unwrap();
96                let took = time_hash(&policy);
97                let params = format!("pbkdf2-sha256 iters={iters}");
98                ladder.push(LadderEntry::new(&params, took, target));
99                consider(&mut best, params, took, target);
100            }
101        }
102    }
103
104    let (selected_params, took) =
105        best.unwrap_or(("(no result)".into(), 0));
106    let distance = took.abs_diff(target);
107
108    if json {
109        // Structured ladder + runner metadata. The ladder lets
110        // operators see the full sweep; the runner block ties results
111        // to the host that produced them so they're not silently
112        // misapplied across heterogeneous fleets.
113        let ladder_json: Vec<serde_json::Value> = ladder
114            .iter()
115            .map(|e| {
116                serde_json::json!({
117                    "candidate": e.candidate,
118                    "measured_ms": e.measured_ms,
119                    "distance_ms": e.distance_ms,
120                    "selected": e.candidate == selected_params,
121                })
122            })
123            .collect();
124        let runner = serde_json::json!({
125            "host_os": std::env::consts::OS,
126            "host_arch": std::env::consts::ARCH,
127            "target_triple": env!("HSH_TARGET_TRIPLE"),
128            "profile": env!("HSH_PROFILE"),
129            "rustc": env!("HSH_RUSTC_VERSION"),
130            "hsh_cli_version": env!("CARGO_PKG_VERSION"),
131        });
132        let ladder_value = serde_json::Value::Array(ladder_json);
133        print_kv(
134            true,
135            &[
136                ("target_ms", &serde_json::Value::from(args.target_ms)),
137                (
138                    "selected_params",
139                    &serde_json::Value::String(selected_params.clone()),
140                ),
141                ("measured_ms", &serde_json::Value::from(took as u64)),
142                (
143                    "distance_ms",
144                    &serde_json::Value::from(distance as u64),
145                ),
146                ("ladder", &ladder_value),
147                ("runner", &runner),
148            ],
149        )?;
150    } else {
151        println!("target:   {} ms", args.target_ms);
152        println!("selected: {selected_params}");
153        println!("measured: {took} ms (off by {distance} ms)");
154        println!("ladder:");
155        for entry in &ladder {
156            let mark = if entry.candidate == selected_params {
157                "*"
158            } else {
159                " "
160            };
161            println!(
162                "  {mark} {} → {} ms (off by {} ms)",
163                entry.candidate, entry.measured_ms, entry.distance_ms
164            );
165        }
166    }
167    Ok(())
168}

170/// One entry in the calibration sweep — a candidate parameter set plus
171/// the measured wall-time and its distance from the target.
172struct LadderEntry {
173    candidate: String,
174    measured_ms: u64,
175    distance_ms: u64,
176}
177
178impl LadderEntry {
179    fn new(candidate: &str, measured: u128, target: u128) -> Self {
180        Self {
181            candidate: candidate.to_owned(),
182            measured_ms: measured as u64,
183            distance_ms: measured.abs_diff(target) as u64,
184        }
185    }
186}
187
188fn time_hash(policy: &Policy) -> u128 {
189    let start = Instant::now();
190    // Best-effort — if hash() errors (e.g. FIPS preset without feature),
191    // record a sentinel so we don't crash the calibrate loop.
192    let _ = hsh::api::hash(policy, PROBE_PASSWORD);
193    start.elapsed().as_millis()
194}
195
196/// Keeps the `(params, took)` whose `took` is closest to `target_ms`.
197/// Ties (`abs_diff` equal) keep the first candidate so the ladder's
198/// lower-cost choice wins — a tighter security upper bound at equal
199/// distance is preferred.
200fn consider(
201    best: &mut Option<(String, u128)>,
202    params: String,
203    took: u128,
204    target_ms: u128,
205) {
206    let new_distance = took.abs_diff(target_ms);
207    match best {
208        None => *best = Some((params, took)),
209        Some((_, current))
210            if new_distance < current.abs_diff(target_ms) =>
211        {
212            *best = Some((params, took));
213        }
214        _ => {}
215    }
216}

218#[cfg(test)]
219mod tests {
220    use super::consider;
221
222    #[test]
223    fn consider_picks_closest_to_target() {
224        let mut best: Option<(String, u128)> = None;
225        // target = 250 ms; ladder produces 100, 200, 300, 500.
226        consider(&mut best, "a".into(), 100, 250);
227        consider(&mut best, "b".into(), 200, 250);
228        consider(&mut best, "c".into(), 300, 250);
229        consider(&mut best, "d".into(), 500, 250);
230        let (chosen, took) = best.unwrap();
231        // 200 and 300 are tied at distance 50; first-wins ⇒ "b".
232        assert_eq!(chosen, "b");
233        assert_eq!(took, 200);
234    }
235
236    #[test]
237    fn consider_keeps_only_candidate() {
238        let mut best: Option<(String, u128)> = None;
239        consider(&mut best, "only".into(), 999, 100);
240        let (chosen, took) = best.unwrap();
241        assert_eq!(chosen, "only");
242        assert_eq!(took, 999);
243    }
244
245    #[test]
246    fn consider_does_not_drift_to_slowest() {
247        // Regression: the prior implementation kept the *largest* took,
248        // so a ladder that exceeded target by a lot would still win.
249        let mut best: Option<(String, u128)> = None;
250        consider(&mut best, "fast".into(), 50, 50);
251        consider(&mut best, "slow".into(), 5_000, 50);
252        let (chosen, _) = best.unwrap();
253        assert_eq!(chosen, "fast");
254    }
255}
