4//! `hsh hash` — produce a storable hash from a password.
5
6use anyhow::{Context, Result};
7
8use crate::cli::HashArgs;
9use crate::commands::resolve_policy;
10use crate::io::{print_kv, resolve_password};
11
12pub(crate) fn run(args: HashArgs, json: bool) -> Result<()> {
13    let password = resolve_password(args.password)
14        .context("resolving password")?;
15    let policy = resolve_policy(args.policy, args.algorithm);
16
17    let stored =
18        hsh::api::hash(&policy, &password).context("hsh::api::hash")?;
19
20    if json {
21        print_kv(
22            true,
23            &[
24                ("stored", &serde_json::Value::String(stored.clone())),
25                (
26                    "algorithm",
27                    &serde_json::Value::String(format!(
28                        "{:?}",
29                        policy.primary()
30                    )),
31                ),
32            ],
33        )?;
34    } else {
35        println!("{stored}");
36    }
37    Ok(())
38}
