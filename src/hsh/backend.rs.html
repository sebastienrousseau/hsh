4//! Backend selector — declares whether the [`crate::Policy`] requires
5//! FIPS 140-3 validated crypto.
6//!
7//! `Backend` is a *requirement* the caller declares; whether it can be
8//! satisfied depends on the build configuration:
9//!
10//! - [`Backend::Native`] — any KDF works; primitives come from the
11//!   pure-Rust RustCrypto stack. **Default.**
12//! - [`Backend::Fips140Required`] — only KDFs with a FIPS 140-3
13//!   validated implementation are allowed. Today that means
14//!   **PBKDF2-HMAC-SHA-256/512** routed through `aws-lc-rs`
15//!   (`fips` feature). Argon2 / bcrypt / scrypt have **no** FIPS
16//!   module anywhere — minting them with `Fips140Required` is a
17//!   compile-time-undetectable error that [`crate::api::hash`] will
18//!   refuse at runtime.
19//!
20//! See `doc/FIPS.md` and `doc/adr/0004-fips-strategy.md` for the full strategy.
21
22use serde::{Deserialize, Serialize};
23
24/// Crypto-validation requirement declared by a [`crate::Policy`].
25#[derive(
26    Clone,
27    Copy,
28    Debug,
29    Default,
30    Eq,
31    Hash,
32    Ord,
33    PartialEq,
34    PartialOrd,
35    Serialize,
36    Deserialize,
37)]
38pub enum Backend {
39    /// Use the pure-Rust RustCrypto primitives. No FIPS claim.
40    #[default]
41    Native,
42    /// Only allow primitives whose underlying implementation is
43    /// FIPS 140-3 validated. Requires the `fips` Cargo feature.
44    Fips140Required,
45}
46
47impl Backend {
48    /// Returns `true` when the backend demands FIPS-validated crypto.
49    #[must_use]
50    pub const fn is_fips(self) -> bool {
51        matches!(self, Self::Fips140Required)
52    }
53
54    /// Returns `true` if this build can actually satisfy a FIPS
55    /// requirement. True when the `fips` Cargo feature is enabled,
56    /// which pulls in the `hsh-backend-awslc` companion crate and
57    /// routes PBKDF2 through AWS-LC FIPS 3.0 (CMVP Cert #4759).
58    /// False otherwise — [`crate::api::hash`] will refuse to mint a
59    /// FIPS-tagged hash when this returns false. See ADR-0004 +
60    /// `doc/FIPS.md`.
61    #[must_use]
62    pub const fn fips_available_in_build() -> bool {
63        cfg!(feature = "fips")
64    }
65}
