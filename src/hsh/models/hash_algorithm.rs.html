7/// The supported password hashing algorithms.
8///
9/// `Argon2id` is the recommended default per RFC 9106 §4. Argon2i is
10/// retained for verifying legacy hashes; Argon2d is exposed for
11/// completeness but rarely the right choice for password hashing.
12#[derive(
13    Clone,
14    Copy,
15    Debug,
16    Eq,
17    Hash,
18    Ord,
19    PartialEq,
20    PartialOrd,
21    Serialize,
22    Deserialize,
23)]
24#[non_exhaustive]
25pub enum HashAlgorithm {
26    /// **Argon2id** — recommended for new password hashes
27    /// (RFC 9106 §4 — hybrid of Argon2i + Argon2d).
28    Argon2id,
29
30    /// **Argon2i** — verify-only for legacy hashes. Has known
31    /// time–memory trade-off attacks when used standalone for password
32    /// hashing; do not use for new hashes.
33    Argon2i,
34
35    /// **Argon2d** — exposed for completeness; vulnerable to
36    /// side-channel attacks. Not recommended for password hashing.
37    Argon2d,
38
39    /// **Bcrypt** — Blowfish-based KDF. 72-byte input ceiling enforced
40    /// by [`crate::algorithms::bcrypt`].
41    Bcrypt,
42
43    /// **Scrypt** — memory-hard KDF. Default params follow OWASP-2025
44    /// (`N = 2^17`, `r = 8`, `p = 1`).
45    Scrypt,
46
47    /// **PBKDF2** — the only KDF with a FIPS 140-3 validated
48    /// implementation (via `aws-lc-rs` behind the `fips` feature).
49    Pbkdf2,
50}

52/// Generic password-hashing trait.
53///
54/// The primary consumer is [`crate::models::hash::Hash`], which uses it
55/// to dispatch to a concrete algorithm.
56pub trait HashingAlgorithm {
57    /// Hashes a plaintext `password` using a specific `salt`.
58    ///
59    /// Returns the raw hash bytes, or a [`crate::error::Error`]
60    /// describing the failure.
61    fn hash_password(password: &str, salt: &str) -> Result<Vec<u8>>;
62}
