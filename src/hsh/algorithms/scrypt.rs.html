4//! Scrypt wrapper with configurable parameters.
5//!
6//! Default parameters follow the OWASP Password Storage Cheat Sheet 2025
7//! minimum recommendation: `N = 2^17 (131 072)`, `r = 8`, `p = 1`,
8//! `dkLen = 64`. The previous default of `N = 2^14` is **below** the
9//! OWASP minimum and should not be used in new deployments.
10
11use crate::error::{Error, Result};
12use crate::models::hash_algorithm::HashingAlgorithm;
13use scrypt::scrypt as scrypt_kdf;
14use scrypt::Params;
15use serde::{Deserialize, Serialize};
16
17/// Default output length in bytes for raw scrypt hashes.
18pub const DEFAULT_OUTPUT_LEN: usize = 64;
19
20/// OWASP-2025 minimum: `log_n = 17 (N = 131 072)`, `r = 8`, `p = 1`,
21/// `dkLen = 64`.
22pub fn owasp_minimum_2025() -> ScryptParams {
23    ScryptParams {
24        log_n: 17,
25        r: 8,
26        p: 1,
27        dk_len: DEFAULT_OUTPUT_LEN,
28    }
29}
30
31/// Scrypt parameters with explicit field names.
32#[derive(Clone, Copy, Debug, Eq, PartialEq, Serialize, Deserialize)]
33pub struct ScryptParams {
34    /// `log2(N)` — the CPU/memory cost factor exponent. OWASP-2025 ≥ 17.
35    pub log_n: u8,
36    /// Block size factor. OWASP-2025 default: 8.
37    pub r: u32,
38    /// Parallelisation factor. OWASP-2025 default: 1.
39    pub p: u32,
40    /// Derived-key length in bytes. Default: 64.
41    pub dk_len: usize,
42}
43
44impl Default for ScryptParams {
45    fn default() -> Self {
46        owasp_minimum_2025()
47    }
48}
49
50impl ScryptParams {
51    /// Converts to the underlying `scrypt::Params`, surfacing parameter
52    /// validation errors.
53    ///
54    /// # Errors
55    ///
56    /// Returns [`Error::InvalidParameter`] if `log_n`, `r`, `p`, or
57    /// `dk_len` violates scrypt's constraints (e.g. `log_n` outside
58    /// `1..64`, or `r * p > 1 << 30`).
59    pub fn to_native(self) -> Result<Params> {
60        Params::new(self.log_n, self.r, self.p, self.dk_len)
61            .map_err(|e| Error::InvalidParameter(e.to_string().into()))
62    }
63}
64
65/// Marker type for the Scrypt hashing algorithm.
66#[derive(
67    Clone,
68    Copy,
69    Debug,
70    Eq,
71    Hash,
72    Ord,
73    PartialEq,
74    PartialOrd,
75    Serialize,
76    Deserialize,
77)]
78pub struct Scrypt;
79
80impl HashingAlgorithm for Scrypt {
81    fn hash_password(password: &str, salt: &str) -> Result<Vec<u8>> {
82        Self::hash_with(password, salt, ScryptParams::default())
83    }
84}
85
86impl Scrypt {
87    /// Hashes `password` with explicit [`ScryptParams`].
88    pub fn hash_with(
89        password: &str,
90        salt: &str,
91        params: ScryptParams,
92    ) -> Result<Vec<u8>> {
93        let native = params.to_native()?;
94        let mut output = vec![0u8; params.dk_len];
95        scrypt_kdf(
96            password.as_bytes(),
97            salt.as_bytes(),
98            &native,
99            &mut output,
100        )
101        .map_err(|e| {
102            Error::hashing(
103                crate::error::HashingErrorKind::Scrypt,
104                e.to_string(),
105            )
106        })?;
107        Ok(output)
108    }
109}
