4//! Versioned `Policy` describing the primary algorithm and per-algorithm
5//! parameters used by the high-level [`crate::api`] surface.
6//!
7//! Construct a policy in one of three ways:
8//!
9//! 1. A **preset** for the common case — [`Policy::owasp_minimum_2025`],
10//!    [`Policy::rfc9106_first_recommended`], [`Policy::fips_140_pbkdf2`].
11//! 2. The **builder** for explicit configuration —
12//!    `PolicyBuilder::new` starting from scratch, or
13//!    `PolicyBuilder::from_preset` starting from a preset and
14//!    overriding select fields.
15//! 3. Combinator methods on a `Policy` for one-off overrides —
16//!    `Policy::with_pepper` (requires the `pepper` feature).
17//!
18//! Fields are non-public; adding new ones is a non-breaking change
19//! per `doc/API-STABILITY.md`.
20

24use crate::algorithms::scrypt::ScryptParams;

26use crate::error::Error;

28#[cfg(feature = "pepper")]

31/// Which algorithm a [`Policy`] uses for *new* hashes.
32///
33/// Verification accepts any of the supported algorithms (Argon2i/d/id,
34/// bcrypt, scrypt, PBKDF2) regardless of the policy's primary.
35#[derive(Clone, Copy, Debug, Eq, Hash, Ord, PartialEq, PartialOrd)]
36#[non_exhaustive]
37pub enum PrimaryAlgorithm {
38    /// **Argon2id** — recommended; the only sensible default for
39    /// new password hashes per RFC 9106 §4.
40    Argon2id,
41    /// **Bcrypt** — useful for ecosystems already on bcrypt.
42    Bcrypt,
43    /// **Scrypt** — memory-hard, simpler parameter story than Argon2.
44    Scrypt,
45    /// **PBKDF2** — the only KDF with a FIPS 140-3 validated path
46    /// (via `aws-lc-rs` behind the `fips` feature). Use this when
47    /// compliance dictates and Argon2id is unavailable.
48    Pbkdf2,
49}
50
51/// Versioned policy snapshot: the algorithm to use for new hashes plus
52/// the parameter ladders for every supported variant.
53///
54/// Construct via [`Policy::owasp_minimum_2025`],
55/// [`Policy::rfc9106_first_recommended`], [`Policy::fips_140_pbkdf2`],
56/// or [`PolicyBuilder`]. Internal fields are non-public; use the
57/// accessor methods for introspection.
58#[derive(Clone, Debug)]
59pub struct Policy {
60    pub(crate) primary: PrimaryAlgorithm,
61    pub(crate) backend: Backend,
62    pub(crate) argon2: Argon2Params,
63    pub(crate) bcrypt: BcryptParams,
64    pub(crate) scrypt: ScryptParams,
65    pub(crate) pbkdf2: Pbkdf2Params,
66    #[cfg(feature = "pepper")]
67    pub(crate) pepper: Option<Arc<dyn hsh_kms::Pepper>>,
68}
69
70impl Policy {
71    /// OWASP Password Storage Cheat Sheet 2025 minimum for all three KDFs.
72    ///
73    /// - **Argon2id**: `m = 19 456 KiB (19 MiB)`, `t = 2`, `p = 1`
74    /// - **Bcrypt**: `cost = 10`, no pre-hash
75    /// - **Scrypt**: `N = 2^17`, `r = 8`, `p = 1`
76    /// - **PBKDF2**: `iters = 600 000`, `dk_len = 32`
77    #[must_use]
78    pub fn owasp_minimum_2025() -> Self {
79        Self {
80            primary: PrimaryAlgorithm::Argon2id,
81            backend: Backend::Native,
82            argon2: argon2id::owasp_minimum_2025(),
83            bcrypt: BcryptParams::new(10),
84            scrypt: ScryptParams::default(),
85            pbkdf2: Pbkdf2Params::owasp_minimum_2025(),
86            #[cfg(feature = "pepper")]
87            pepper: None,
88        }
89    }
90
91    /// RFC 9106 §4 first-recommended for Argon2id: `m = 2^21 (2 GiB)`,
92    /// `t = 1`, `p = 4`. Bcrypt and scrypt fall back to OWASP-2025.
93    #[must_use]
94    pub fn rfc9106_first_recommended() -> Self {
95        Self {
96            primary: PrimaryAlgorithm::Argon2id,
97            backend: Backend::Native,
98            argon2: argon2id::rfc9106_first_recommended(),
99            bcrypt: BcryptParams::new(10),
100            scrypt: ScryptParams::default(),
101            pbkdf2: Pbkdf2Params::owasp_minimum_2025(),
102            #[cfg(feature = "pepper")]
103            pepper: None,
104        }
105    }
106
107    /// FIPS 140-3 deployment preset: **PBKDF2-HMAC-SHA-256, 600 000
108    /// iterations** (OWASP-2025 minimum), with
109    /// [`Backend::Fips140Required`].
110    ///
111    /// Argon2 / bcrypt / scrypt remain present in this preset's
112    /// parameter ladder for verifying legacy hashes, but
113    /// [`crate::api::hash`] will refuse to mint new hashes under those
114    /// algorithms when `backend == Backend::Fips140Required` because
115    /// they have no FIPS-validated implementation anywhere.
116    #[must_use]
117    pub fn fips_140_pbkdf2() -> Self {
118        Self {
119            primary: PrimaryAlgorithm::Pbkdf2,
120            backend: Backend::Fips140Required,
121            argon2: argon2id::owasp_minimum_2025(),
122            bcrypt: BcryptParams::new(10),
123            scrypt: ScryptParams::default(),
124            pbkdf2: Pbkdf2Params::owasp_minimum_2025(),
125            #[cfg(feature = "pepper")]
126            pepper: None,
127        }
128    }
129
130    /// Returns the primary algorithm new hashes are minted under.
131    #[must_use]
132    pub const fn primary(&self) -> PrimaryAlgorithm {
133        self.primary
134    }
135
136    /// Returns the crypto-validation requirement.
137    #[must_use]
138    pub const fn backend(&self) -> Backend {
139        self.backend
140    }
141
142    /// Returns a reference to the Argon2 parameter set.
143    #[must_use]
144    pub const fn argon2_params(&self) -> &Argon2Params {
145        &self.argon2
146    }
147
148    /// Returns the bcrypt parameter set.
149    #[must_use]
150    pub const fn bcrypt_params(&self) -> BcryptParams {
151        self.bcrypt
152    }
153
154    /// Returns the scrypt parameter set.
155    #[must_use]
156    pub const fn scrypt_params(&self) -> ScryptParams {
157        self.scrypt
158    }
159
160    /// Returns the PBKDF2 parameter set.
161    #[must_use]
162    pub const fn pbkdf2_params(&self) -> Pbkdf2Params {
163        self.pbkdf2
164    }
165
166    /// Returns whether a pepper provider is attached. Always `false`
167    /// without the `pepper` feature.
168    #[must_use]
169    pub fn has_pepper(&self) -> bool {
170        #[cfg(feature = "pepper")]
171        {
172            self.pepper.is_some()
173        }
174        #[cfg(not(feature = "pepper"))]
175        {
176            false
177        }
178    }
179
180    /// Attaches a pepper provider to this policy and returns the
181    /// updated value.
182    ///
183    /// Accepts any `impl hsh_kms::Pepper + 'static` so callers don't
184    /// have to wrap their provider in `Arc` manually — the wrap is
185    /// applied internally. Pass an already-wrapped `Arc<dyn Pepper>`
186    /// via [`Self::with_pepper_arc`] when sharing across instances.
187    ///
188    /// New hashes will be peppered with `provider.current()`. Old
189    /// hashes carrying earlier key versions remain verifiable, and
190    /// [`crate::api::verify_and_upgrade`] will signal `needs_rehash`
191    /// when an old version is detected.
192    #[cfg(feature = "pepper")]
193    #[must_use]
194    pub fn with_pepper(
195        mut self,
196        provider: impl hsh_kms::Pepper + 'static,
197    ) -> Self {
198        self.pepper = Some(Arc::new(provider));
199        self
200    }
201
202    /// Variant of [`Self::with_pepper`] that accepts an already-wrapped
203    /// `Arc<dyn Pepper>` — useful when the same provider instance is
204    /// shared across multiple policies.
205    #[cfg(feature = "pepper")]
206    #[must_use]
207    pub fn with_pepper_arc(
208        mut self,
209        provider: Arc<dyn hsh_kms::Pepper>,
210    ) -> Self {
211        self.pepper = Some(provider);
212        self
213    }
214
215    /// Returns a [`PolicyBuilder`] seeded with the fields of this
216    /// policy, suitable for forking + overriding select values.
217    #[must_use]
218    pub fn to_builder(&self) -> PolicyBuilder {
219        PolicyBuilder::from_preset(self)
220    }
221
222    /// Returns `true` if `stored_params` are at least as strong as the
223    /// current policy's Argon2 params. Used by [`crate::api::verify_and_upgrade`]
224    /// to decide whether a successful verify should trigger a rehash.
225    pub(crate) fn argon2_satisfies(
226        &self,
227        stored: &Argon2Params,
228    ) -> bool {
229        stored.m_cost() >= self.argon2.m_cost()
230            && stored.t_cost() >= self.argon2.t_cost()
231            && stored.p_cost() >= self.argon2.p_cost()
232            && stored.output_len() == self.argon2.output_len()
233    }
234
235    /// Returns `true` if a stored bcrypt `cost` is at least as strong as
236    /// the current policy. A weaker stored cost triggers a rehash on
237    /// successful verify.
238    pub(crate) fn bcrypt_satisfies(&self, stored_cost: u32) -> bool {
239        stored_cost >= self.bcrypt.cost
240    }
241
242    /// Returns `true` if stored scrypt `log_n`, `r`, `p`, and `dk_len`
243    /// are at least as strong as the current policy. Cost factors must
244    /// be `>=`; `dk_len` must match exactly (changing output length
245    /// changes the meaning of the stored bytes).
246    pub(crate) fn scrypt_satisfies(
247        &self,
248        stored: &ScryptParams,
249    ) -> bool {
250        stored.log_n >= self.scrypt.log_n
251            && stored.r >= self.scrypt.r
252            && stored.p >= self.scrypt.p
253            && stored.dk_len == self.scrypt.dk_len
254    }
255
256    /// Returns `true` if stored PBKDF2 `iterations` and `dk_len` are at
257    /// least as strong as the current policy. PRF equality is checked
258    /// by the caller via `algo_id` matching.
259    pub(crate) fn pbkdf2_satisfies(
260        &self,
261        stored_iters: u32,
262        stored_dk_len: usize,
263    ) -> bool {
264        stored_iters >= self.pbkdf2.iterations
265            && stored_dk_len == self.pbkdf2.dk_len
266    }
267}
268
269impl Default for Policy {
270    fn default() -> Self {
271        Self::owasp_minimum_2025()
272    }
273}

275/// Fluent builder for [`Policy`].
276///
277/// Construct via [`PolicyBuilder::new`] for a blank slate (in which
278/// case [`PolicyBuilder::build`] requires at least `primary`), or via
279/// [`PolicyBuilder::from_preset`] to start from one of the presets and
280/// override individual fields.
281///
282/// ## Example
283///
284/// ```
285/// use hsh::policy::{Policy, PolicyBuilder, PrimaryAlgorithm};
286///
287/// let policy = PolicyBuilder::from_preset(&Policy::owasp_minimum_2025())
288///     .primary(PrimaryAlgorithm::Pbkdf2)
289///     .build()
290///     .expect("valid policy");
291///
292/// assert_eq!(policy.primary(), PrimaryAlgorithm::Pbkdf2);
293/// ```
294#[derive(Clone, Debug, Default)]
295pub struct PolicyBuilder {
296    primary: Option<PrimaryAlgorithm>,
297    backend: Option<Backend>,
298    argon2: Option<Argon2Params>,
299    bcrypt: Option<BcryptParams>,
300    scrypt: Option<ScryptParams>,
301    pbkdf2: Option<Pbkdf2Params>,
302    #[cfg(feature = "pepper")]
303    pepper: Option<Arc<dyn hsh_kms::Pepper>>,
304}
305
306impl PolicyBuilder {
307    /// Starts a blank builder. [`build`](Self::build) will error if at
308    /// least the primary algorithm isn't set.
309    #[must_use]
310    pub fn new() -> Self {
311        Self::default()
312    }
313
314    /// Seeds the builder from an existing policy so individual fields
315    /// can be overridden without re-listing the others.
316    #[must_use]
317    pub fn from_preset(policy: &Policy) -> Self {
318        Self {
319            primary: Some(policy.primary),
320            backend: Some(policy.backend),
321            argon2: Some(policy.argon2.clone()),
322            bcrypt: Some(policy.bcrypt),
323            scrypt: Some(policy.scrypt),
324            pbkdf2: Some(policy.pbkdf2),
325            #[cfg(feature = "pepper")]
326            pepper: policy.pepper.clone(),
327        }
328    }
329
330    /// Sets the primary algorithm new hashes are minted under.
331    #[must_use]
332    pub fn primary(mut self, primary: PrimaryAlgorithm) -> Self {
333        self.primary = Some(primary);
334        self
335    }
336
337    /// Sets the crypto-validation requirement (FIPS / Native).
338    #[must_use]
339    pub fn backend(mut self, backend: Backend) -> Self {
340        self.backend = Some(backend);
341        self
342    }
343
344    /// Overrides the Argon2 parameter set.
345    #[must_use]
346    pub fn argon2(mut self, params: Argon2Params) -> Self {
347        self.argon2 = Some(params);
348        self
349    }
350
351    /// Overrides the bcrypt parameter set.
352    #[must_use]
353    pub fn bcrypt(mut self, params: BcryptParams) -> Self {
354        self.bcrypt = Some(params);
355        self
356    }
357
358    /// Overrides the scrypt parameter set.
359    #[must_use]
360    pub fn scrypt(mut self, params: ScryptParams) -> Self {
361        self.scrypt = Some(params);
362        self
363    }
364
365    /// Overrides the PBKDF2 parameter set.
366    #[must_use]
367    pub fn pbkdf2(mut self, params: Pbkdf2Params) -> Self {
368        self.pbkdf2 = Some(params);
369        self
370    }
371
372    /// Attaches a pepper provider. Requires the `pepper` feature.
373    ///
374    /// Accepts any `impl Pepper + 'static`; the `Arc` wrap is internal.
375    #[cfg(feature = "pepper")]
376    #[must_use]
377    pub fn pepper(
378        mut self,
379        provider: impl hsh_kms::Pepper + 'static,
380    ) -> Self {
381        self.pepper = Some(Arc::new(provider));
382        self
383    }
384
385    /// Variant of [`Self::pepper`] for callers holding an already-
386    /// wrapped `Arc<dyn Pepper>` (e.g. sharing across many builders).
387    #[cfg(feature = "pepper")]
388    #[must_use]
389    pub fn pepper_arc(
390        mut self,
391        provider: Arc<dyn hsh_kms::Pepper>,
392    ) -> Self {
393        self.pepper = Some(provider);
394        self
395    }
396
397    /// Removes any attached pepper provider. Requires the `pepper`
398    /// feature.
399    #[cfg(feature = "pepper")]
400    #[must_use]
401    pub fn no_pepper(mut self) -> Self {
402        self.pepper = None;
403        self
404    }
405
406    /// Finalises the builder into a [`Policy`].
407    ///
408    /// # Errors
409    ///
410    /// Returns [`Error::InvalidPolicy`] if the builder was started
411    /// from [`new`](Self::new) without calling [`primary`](Self::primary).
412    pub fn build(self) -> Result<Policy, Error> {
413        Ok(Policy {
414            primary: self.primary.ok_or(Error::InvalidPolicy(
415                "primary algorithm required".into(),
416            ))?,
417            backend: self.backend.unwrap_or_default(),
418            argon2: self
419                .argon2
420                .unwrap_or_else(argon2id::owasp_minimum_2025),
421            bcrypt: self
422                .bcrypt
423                .unwrap_or_else(|| BcryptParams::new(10)),
424            scrypt: self.scrypt.unwrap_or_default(),
425            pbkdf2: self.pbkdf2.unwrap_or_default(),
426            #[cfg(feature = "pepper")]
427            pepper: self.pepper,
428        })
429    }
