4//! Argument parsing for `hsh-cli`.
5
6use anyhow::Result;
7use clap::{Parser, Subcommand, ValueEnum};
8
9/// `hsh` — password hashing on the command line.
10#[derive(Debug, Parser)]
11#[command(
12    name = "hsh",
13    version,
14    about = "Enterprise password hashing for the command line.",
15    long_about = None,
16)]
17pub(crate) struct Cli {
18    /// Emit machine-readable JSON instead of plain text.
19    #[arg(long, global = true)]
20    pub json: bool,
21
22    #[command(subcommand)]
23    pub command: Command,
24}
25
26/// Top-level subcommands.
27#[derive(Debug, Subcommand)]
28pub(crate) enum Command {
29    /// Hash a password and print the storable form to stdout.
30    Hash(HashArgs),
31    /// Verify a candidate password against a stored hash.
32    Verify(VerifyArgs),
33    /// Verify, then mint a fresh hash under the current policy.
34    Rehash(RehashArgs),
35    /// Pretty-print the algorithm + parameters of a stored hash.
36    Inspect(InspectArgs),
37    /// Show the effective crypto route for a preset (operator self-check).
38    #[command(
39        long_about = "Show the effective crypto route for a given preset: \
40                      which Backend the policy declares, whether this build \
41                      can satisfy a FIPS requirement, the primary algorithm \
42                      new hashes are minted under, whether the `pepper` \
43                      feature is compiled in, and build provenance (hsh-cli \
44                      version, rustc, target triple, profile). Use this \
45                      before a deployment takes traffic to verify the binary \
46                      actually delivers the contract you expected."
47    )]
48    InspectBackend(InspectBackendArgs),
49    /// Calibrate KDF parameters to hit a wall-time target.
50    Calibrate(CalibrateArgs),
51    /// Emit shell-completion scripts for the named shell.
52    Completions(CompletionsArgs),
53}
54
55/// Selectable preset policy.
56#[derive(Debug, Clone, Copy, ValueEnum, Default)]
57pub(crate) enum PresetPolicy {
58    /// OWASP-2025 minimum (Argon2id, m=19456, t=2, p=1).
59    #[default]
60    Owasp,
61    /// RFC 9106 §4 first-recommended (Argon2id, m=2^21, t=1, p=4).
62    Rfc9106,
63    /// Hardened FIPS profile (PBKDF2-HMAC-SHA-256, 600k iters,
64    /// Backend::Fips140Required — requires a FIPS-capable build).
65    Fips,
66}
67
68/// Algorithm tag accepted on the command line.
69#[derive(Debug, Clone, Copy, ValueEnum)]
70pub(crate) enum AlgoArg {
71    /// Argon2id — recommended default.
72    Argon2id,
73    /// Argon2i — verify-only for legacy hashes.
74    Argon2i,
75    /// Argon2d — exposed for completeness.
76    Argon2d,
77    /// Bcrypt — Blowfish-based KDF.
78    Bcrypt,
79    /// Scrypt — memory-hard KDF.
80    Scrypt,
81    /// PBKDF2-HMAC-SHA-256 — the only FIPS-validated path.
82    Pbkdf2,
83}
84
85#[derive(Debug, Clone, clap::Args)]
86pub(crate) struct HashArgs {
87    /// Preset policy to apply. Defaults to OWASP-2025.
88    #[arg(long, value_enum, default_value_t)]
89    pub policy: PresetPolicy,
90    /// Override the primary algorithm.
91    #[arg(short, long, value_enum)]
92    pub algorithm: Option<AlgoArg>,
93    /// Password (insecure: leaves password in shell history).
94    /// Omit and provide via stdin or `$HSH_PASSWORD` instead.
95    #[arg(long, env = "HSH_PASSWORD", hide_env_values = true)]
96    pub password: Option<String>,
97}
98
99#[derive(Debug, Clone, clap::Args)]
100pub(crate) struct VerifyArgs {
101    /// Stored hash string (PHC / MCF / hsh-pepper:…).
102    #[arg(short = 'H', long, env = "HSH_STORED")]
103    pub stored: String,
104    /// Preset policy to apply for rehash detection.
105    #[arg(long, value_enum, default_value_t)]
106    pub policy: PresetPolicy,
107    /// Password (insecure: leaves password in shell history).
108    #[arg(long, env = "HSH_PASSWORD", hide_env_values = true)]
109    pub password: Option<String>,
110}
111
112#[derive(Debug, Clone, clap::Args)]
113pub(crate) struct RehashArgs {
114    /// Stored hash string.
115    #[arg(short = 'H', long, env = "HSH_STORED")]
116    pub stored: String,
117    /// Preset policy to mint the new hash under.
118    #[arg(long, value_enum, default_value_t)]
119    pub policy: PresetPolicy,
120    /// Password (insecure).
121    #[arg(long, env = "HSH_PASSWORD", hide_env_values = true)]
122    pub password: Option<String>,
123}
124
125#[derive(Debug, Clone, clap::Args)]
126pub(crate) struct InspectArgs {
127    /// Stored hash string to inspect.
128    pub hash: String,
129}
130
131#[derive(Debug, Clone, clap::Args)]
132pub(crate) struct InspectBackendArgs {
133    /// Preset to evaluate. Defaults to OWASP-2025.
134    #[arg(long, value_enum, default_value_t)]
135    pub policy: PresetPolicy,
136}
137
138#[derive(Debug, Clone, clap::Args)]
139pub(crate) struct CalibrateArgs {
140    /// Algorithm to calibrate.
141    #[arg(short, long, value_enum, default_value_t = AlgoArg::Argon2id)]
142    pub algorithm: AlgoArg,
143    /// Target wall-time per `hash` in milliseconds.
144    #[arg(short = 't', long, default_value_t = 500)]
145    pub target_ms: u32,
146}
147
148#[derive(Debug, Clone, clap::Args)]
149pub(crate) struct CompletionsArgs {
150    /// Shell to emit completions for.
151    #[arg(value_enum)]
152    pub shell: clap_complete::Shell,
153}
154
155impl Cli {
156    /// Dispatches to the chosen subcommand.
157    pub(crate) fn run(self) -> Result<()> {
158        match self.command {
159            Command::Hash(args) => {
160                crate::commands::hash::run(args, self.json)
161            }
162            Command::Verify(args) => {
163                crate::commands::verify::run(args, self.json)
164            }
165            Command::Rehash(args) => {
166                crate::commands::rehash::run(args, self.json)
167            }
168            Command::Inspect(args) => {
169                crate::commands::inspect::run(args, self.json)
170            }
171            Command::InspectBackend(args) => {
172                crate::commands::inspect_backend::run(args, self.json)
173            }
174            Command::Calibrate(args) => {
175                crate::commands::calibrate::run(args, self.json)
176            }
177            Command::Completions(args) => {
178                crate::commands::completions::run(args)
179            }
180        }
181    }
182}
